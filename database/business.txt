data_business (business details and info database table)
-------------------

id (int) (auto) (primary),

org_gatuadress (string/text) (default NULL),
org_postadress (string/text) (default NULL),
org_postnummer (string/text) (default NULL),
org_postort (string/text) (default NULL),

org_organisationsnummer (string/text) (default NULL),
org_bolagsform (string/text) (default NULL),
org_sate (string/text) (default NULL),
org_status (string/text) (default NULL),

org_bildat (date time) (default TIMESTAMP),
org_registrerat (date time) (default TIMESTAMP),
org_registreringsdatum (date time) (default TIMESTAMP),
org_bolagsordning (date time) (default TIMESTAMP),

is_active [data record is active] (boolean) (default TRUE),

created_at [data record created at] (date time) (default TIMESTAMP),
updated_at [data record updated at] (date time) (default TIMESTAMP),
